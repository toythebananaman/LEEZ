to save:
git add . ;git commit -m "bananas";git push origin branchname

to create a git/hub connection:
git init;git add . ;git commit -m "start";git remote add origin https://github.com/OWNER/REPOSITORY.git;git push origin main/master

create a new branch:
git checkout -b branch name

to switch branch:
git checkout branchname

git alias:
git config --global alias.SHORTCUT 'COMMAND'

to show graph:
git log --all --graph

or if setup alias:
git lg

remove git remote
git remote | while read -r remote; do git remote remove "$remote"; done

remove git entirely
rm -rf .git*

in gitignore:
.DS_Store
Thumbs.db
desktop.ini
*.suo
*.user
.vscode/

to delete all the files to delete, for mac and windws and linux:
find . -name .DS_Store -print0 | xargs -0 git rm -f --ignore-unmatch
find . -name Thumbs.db -print0 | xargs -0 git rm -f --ignore-unmatch
find . -name desktop.ini -print0 | xargs -0 git rm -f --ignore-unmatch
find . -name '*.suo' -print0 | xargs -0 git rm -f --ignore-unmatch
find . -name '*.user' -print0 | xargs -0 git rm -f --ignore-unmatch
find . -type d -name .vscode -print0 | xargs -0 git rm -rf --ignore-unmatch
